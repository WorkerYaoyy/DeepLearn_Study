# Anaconda使用笔记

```bash
#查看所有虚拟环境
conda env list
# 等价简写
conda info --envs
# 创建指定Python版本的环境，命名myenv
conda create -n myenv python=3.8
# 创建时同时预装包
conda create -n myenv python=3.8 numpy pillow
#激活进入环境
conda activate myenv
# 退出当前环境，回到base
conda deactivate
#删除myenv环境
conda remove -n myenv --all
# 确认删除输入y
#复制克隆环境
conda create -n new_env --clone old_env
#查看当前环境已装 conda 包
conda list
# 过滤torch相关包
conda list torch
# 查看conda版本
conda --version
# 查看conda完整信息（Python路径、环境位置）
conda info
# 查看当前环境Python版本
python --version
# 查看CUDA驱动版本（NVIDIA显卡）
nvidia-smi
```
